async function loadData() { try { const response = await fetch('Market_Analysis/scanned_laws.json?t=' + Date.now()); const data = await response.json(); const dashboard = document.getElementById('dashboard'); dashboard.innerHTML = ''; // ניקוי הלוח let lawCounter = 1; // מונה חוקים רץ for (const [site, info] of Object.entries(data)) { if (site === "" || !info) continue; // דילוג על אתרים ריקים // איסוף מכל השדות האפשריים שה-AI יצר const rawIssues = info.security_flaws || info.vulnerabilities || info.potential_technical_flaws || []; if (rawIssues.length === 0) continue; const card = document.createElement('div'); card.className = 'col-12 mb-4'; // כרטיס רחב לכל אתר const lawsHtml = rawIssues.map(f => { const id = f.law_id || `CL-${100 + lawCounter++}`; // אם אין ID, מייצר אחד const title = f.flaw || "חוק הגנת המידע"; const desc = f.description || (f.fix ? f.fix.problem : "חובה לתקן ליקוי אבטחה זה באופן מיידי."); const sevColor = f.severity === 'High' ? '#dc3545' : '#ffc107'; return `
${id} ${f.severity || 'Medium'}
${title}

${desc}

`; }).join(''); card.innerHTML = `
🌐 מקור החוק: ${site}
${lawsHtml}
`; dashboard.appendChild(card); } } catch (e) { console.error("Error:", e); } }
